# HTML基本布局与字符实体

# 【iframe】内联框架

    <iframe src="URL" name="框架名"></iframe>

# 【width】定义宽度

# 【height】定义高度

# 【frameborder】

规定是否显示 iframe 周围的边框。

提示

设置属性值为 "0"就可以移除边框。

# 【target】

可用作链接的目标 链接的 target 属性必须引用 iframe 的 name属性。

# 【head】头部元素

# 【head】定义关于文档的信息。

     <head></head>

# 【title】定义文档标题。

    <title></title>

# 【base】定义页面上所有链接的默认地址或默认目标。

    <base href="">

# 【link】定义文档与外部资源之间的关系。

    <link rel="stylesheet" href="">

一般用于连接外部样式表。

# 【meta】定义关于 HTML 文档的元数据。

<meta>标签里面存放的都是一些规则。

 <meta charset="UTF-8"> 解决中文乱码问题，utf-8即万国码

# 【script】定义客户端脚本。(JavaScript)

    <script></script>

# 【style】定义文档的样式信息。（css）[内部样式]

    <style></style>

# 【HTML5语义元素】

# 【header】定义文档或节的页眉

    <header></header>

# 【nav】定义导航链接的容器

    <nav></nav>

# 【section】定义文档中的节

    <section></section>

# 【article】定义独立的自包含文章

    <article></article>

# 【aside】定义内容之外的内容（比如侧栏）

    <aside></aside>

# 【footer】定义文档或节的页脚

    <footer></footer>

# 【details】定义额外的细节

    <details></details>

# 【summary】定义 details 元素的标题

     <summary></summary>

# 【figure】规定自包含内容，比如图示、图表、照片、代码清单等。

    <figure></figure>

# 【figcaption】定义<figure>元素的标题。

    <figcaption></figcaption>

# 【main】规定文档的主内容。

    <main></main>

# 【progress】定义任务进度。

    <progress></progress>
